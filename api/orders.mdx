The Orders API provides endpoints for retrieving and managing Shopify orders.

## Poll Recent Orders

Retrieves the most recent orders from Shopify with detailed information including line items, customer data, and shipping addresses.

<CodeGroup>
```http GET
GET /api/orders/poll
```

```http POST
POST /api/orders/poll
```
</CodeGroup>

<Note>
  Both GET and POST methods are supported for this endpoint.
</Note>

### Response

<ResponseField name="success" type="boolean">
  Whether the request was successful
</ResponseField>

<ResponseField name="orders" type="array">
  Array of recent orders (up to 10)
  
  <Expandable title="Order Object">
    <ResponseField name="id" type="string">
      Shopify order ID (GraphQL format)
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Order name (e.g., "#1001")
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      Order creation timestamp (ISO 8601)
    </ResponseField>
    
    <ResponseField name="financialStatus" type="string">
      Financial status (PENDING, AUTHORIZED, PAID, etc.)
    </ResponseField>
    
    <ResponseField name="fulfillmentStatus" type="string">
      Fulfillment status (UNFULFILLED, PARTIAL, FULFILLED, etc.)
    </ResponseField>
    
    <ResponseField name="totalPrice" type="string">
      Total order amount
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      Currency code (e.g., "USD", "EUR")
    </ResponseField>
    
    <ResponseField name="lineItems" type="array">
      Order line items
      
      <Expandable title="Line Item">
        <ResponseField name="id" type="string">
          Line item ID
        </ResponseField>
        
        <ResponseField name="name" type="string">
          Product name
        </ResponseField>
        
        <ResponseField name="quantity" type="integer">
          Quantity ordered
        </ResponseField>
        
        <ResponseField name="variant" type="string">
          Variant title
        </ResponseField>
        
        <ResponseField name="sku" type="string">
          SKU code
        </ResponseField>
        
        <ResponseField name="productId" type="string">
          Product ID
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="shippingAddress" type="object">
      Shipping address details
      
      <Expandable title="Address Fields">
        <ResponseField name="firstName" type="string">
          First name
        </ResponseField>
        
        <ResponseField name="lastName" type="string">
          Last name
        </ResponseField>
        
        <ResponseField name="address1" type="string">
          Address line 1
        </ResponseField>
        
        <ResponseField name="city" type="string">
          City
        </ResponseField>
        
        <ResponseField name="province" type="string">
          State/Province
        </ResponseField>
        
        <ResponseField name="country" type="string">
          Country
        </ResponseField>
        
        <ResponseField name="zip" type="string">
          Postal code
        </ResponseField>
        
        <ResponseField name="phone" type="string">
          Phone number
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="customer" type="object">
      Customer information
      
      <Expandable title="Customer Fields">
        <ResponseField name="id" type="string">
          Customer ID
        </ResponseField>
        
        <ResponseField name="email" type="string">
          Email address
        </ResponseField>
        
        <ResponseField name="firstName" type="string">
          First name
        </ResponseField>
        
        <ResponseField name="lastName" type="string">
          Last name
        </ResponseField>
        
        <ResponseField name="phone" type="string">
          Phone number
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="tags" type="array">
      Order tags
    </ResponseField>
    
    <ResponseField name="note" type="string">
      Order note
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="integer">
  Number of orders returned
</ResponseField>

</Note>
